\chapter{Module de Saisie Utilisateur (Inputs)}

\section{Introduction}
Ce module est le pont entre l'utilisateur et la logique de notre planificateur de transport. Il regroupe les champs de texte (\textit{Entries}), les listes déroulantes (\textit{Drop-downs}) et les sélecteurs numériques (\textit{Spin Buttons}). Ces composants valident et transmettent les données saisies à notre application.

\section{Structures et Définitions (Header)}

\begin{inputbox}{Structure : entry\_config}
La structure \texttt{entry\_config} définit le comportement des champs de saisie. Elle supporte le masquage des caractères (pour les mots de passe) et des indices visuels (\textit{placeholders}).
\end{inputbox}

\begin{lstlisting}
typedef struct {
    const char *placeholder;
    const char *default_text;
    const char *purpose_hint;
    bool password;
    bool editable;
    int max_length;
    widget_style_config style;
    void (*on_changed)(GtkEditable *editable, gpointer user_data);
    gpointer user_data;
} entry_config;
\end{lstlisting}

\section{Logique de Traitement (Source)}

\begin{inputbox}{Analyse de l'Intention : parse\_input\_purpose}
Cette fonction statique est une aide précieuse. Elle convertit une simple chaîne de caractères (ex: "email") en une énumération \texttt{GtkInputPurpose}. Cela permet d'adapter le clavier virtuel sur mobile ou de valider le format de saisie.
\end{inputbox}

\begin{lstlisting}
static GtkInputPurpose parse_input_purpose(const char *hint) {
    if (hint == NULL) return GTK_INPUT_PURPOSE_FREE_FORM;
    if (strcmp(hint, "email") == 0) return GTK_INPUT_PURPOSE_EMAIL;
    if (strcmp(hint, "number") == 0) return GTK_INPUT_PURPOSE_NUMBER;
    // ... selection du type de saisie ...
    return GTK_INPUT_PURPOSE_FREE_FORM;
}
\end{lstlisting}

\section{Composants de Sélection}

\begin{inputbox}{Fonction : create\_dropdown}
Le \texttt{GtkDropDown} remplace l'ancienne \texttt{ComboBox}. Notre wrapper simplifie sa création à partir d'un tableau de chaînes de caractères et permet d'activer la recherche interne pour les listes longues (comme une liste de villes au Maroc).
\end{inputbox}

\begin{lstlisting}
GtkWidget *create_dropdown(const dropdown_config *config) {
    GtkWidget *dropdown;
    if (config == NULL || config->options == NULL) {
        return gtk_drop_down_new(NULL, NULL);
    }

    dropdown = gtk_drop_down_new_from_strings(config->options);
    gtk_drop_down_set_enable_search(GTK_DROP_DOWN(dropdown), config->enable_search);
    
    if (config->on_selected) {
        g_signal_connect(dropdown, "notify::selected", G_CALLBACK(config->on_selected), config->user_data);
    }

    apply_widget_style(dropdown, &config->style);
    return dropdown;
}
\end{lstlisting}

\section{Ajustements Numériques}

\begin{inputbox}{Fonction : create\_spin\_button}
Idéal pour choisir le nombre de passagers ou l'heure, le \texttt{SpinButton} utilise un \texttt{GtkAdjustment} pour définir ses bornes. 
On définit la valeur actuelle $v$, le minimum $v_{min}$ et le maximum $v_{max}$.
\end{inputbox}

\begin{lstlisting}
GtkWidget *create_spin_button(const spin_button_config *config) {
    GtkAdjustment *adj;
    adj = gtk_adjustment_new(config->value, config->min_value, config->max_value, config->step, config->page, 0.0);
    GtkWidget *spin = gtk_spin_button_new(adj, config->step, config->digits);
    
    apply_widget_style(spin, &config->style);
    return spin;
}
\end{lstlisting}
