\section{Éléments de saisie (Input)}

\subsection{Structure \texttt{entry\_config}}
\begin{lstlisting}[language=C]
typedef struct {
    const char *placeholder;
    const char *default_text;
    const char *purpose_hint;
    bool password;
    bool editable;
    int max_length;
    widget_style_config style;
    void (*on_changed)(GtkEditable *editable, gpointer user_data);
    gpointer user_data;
} entry_config;
\end{lstlisting}

\subsection{Description des champs (Entry)}
\begin{center}
\begin{tabular}{|l|l|p{5cm}|}
\hline
\textbf{Champ} & \textbf{Type} & \textbf{Description} \\
\hline
placeholder & const char* & Texte indicatif quand le champ est vide \\
\hline
default\_text & const char* & Texte pré-rempli par défaut \\
\hline
password & bool & Masque les caractères saisis \\
\hline
editable & bool & Permet la modification par l'utilisateur \\
\hline
max\_length & int & Longueur maximale de la saisie \\
\hline
on\_changed & callback & Fonction appelée à chaque modification \\
\hline
\end{tabular}
\end{center}

\subsection{Structure \texttt{dropdown\_config}}
\begin{lstlisting}[language=C]
typedef struct {
    const char **options;
    guint selected_index;
    bool enable_search;
    widget_style_config style;
    void (*on_selected)(GtkDropDown *dropdown, GParamSpec *pspec, gpointer user_data);
    gpointer user_data;
} dropdown_config;
\end{lstlisting}

\begin{center}
\begin{tabular}{|l|l|p{5cm}|}
\hline
\textbf{Champ} & \textbf{Type} & \textbf{Description} \\
\hline
options & const char** & Tableau de chaînes de caractères (choix) \\
\hline
selected\_index & guint & Index de l'élément sélectionné par défaut \\
\hline
enable\_search & bool & Active la barre de recherche intégrée \\
\hline
on\_selected & callback & Fonction appelée lors d'un changement de sélection \\
\hline
\end{tabular}
\end{center}

\subsection{Exemple d'utilisation}
\begin{lstlisting}[language=C]
#include "input.h"

void create_inputs(GtkWidget *box) {
    entry_config entry = {
        .placeholder = "Entrez votre nom...",
        .password = false,
        .editable = true
    };
    gtk_box_append(GTK_BOX(box), create_entry(&entry));
}
\end{lstlisting}
\subsection{Structure \texttt{spin\_button\_config}}
\begin{lstlisting}[language=C]
typedef struct {
    double min_value;
    double max_value;
    double step;
    double page;
    double value;
    guint digits;
    bool numeric_only;
    bool wrap;
    widget_style_config style;
    void (*on_value_changed)(GtkSpinButton *spin, gpointer user_data);
    gpointer user_data;
} spin_button_config;
\end{lstlisting}
\begin{center}
\begin{tabular}{|l|l|>{\raggedright\arraybackslash}p{6cm}|}
\hline
\textbf{Champ} & \textbf{Type} & \textbf{Description} \\
\hline
min\_value / max\_value & double & Limites de la valeur \\
\hline
step / page & double & Incréments (flèches et saut de page) \\
\hline
value & double & Valeur par défaut \\
\hline
digits & guint & Nombre de décimales à afficher \\
\hline
numeric\_only & bool & Restreint la saisie aux chiffres \\
\hline
wrap & bool & Recommence au minimum une fois le maximum atteint \\
\hline
on\_value\_changed & callback & Fonction appelée au changement de valeur \\
\hline
\end{tabular}
\end{center}
