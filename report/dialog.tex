\section{Boîtes de dialogue (Dialog)}

\subsection{Structure \texttt{alert\_dialog\_config}}
\begin{lstlisting}[language=C]
typedef struct {
    GtkWindow *parent;
    const char *message;
    const char *detail;
    const alert_button_config *buttons;
    size_t button_count;
    int default_button;
    int cancel_button;
    bool modal;
    void (*on_response)(int response_index, gpointer user_data);
    gpointer user_data;
} alert_dialog_config;
\end{lstlisting}

\subsection{Description des champs}
\begin{center}
\begin{tabular}{|l|l|p{5cm}|}
\hline
\textbf{Champ} & \textbf{Type} & \textbf{Description} \\
\hline
parent & GtkWindow* & Fenêtre parente de l'alerte \\
\hline
message & const char* & Titre ou message principal \\
\hline
detail & const char* & Texte secondaire détaillé \\
\hline
buttons & alert\_button\_config* & Tableau des boutons à afficher \\
\hline
on\_response & callback & Fonction appelée après un choix \\
\hline
\end{tabular}
\end{center}
\subsection{Structure \texttt{alert\_button\_config}}
\begin{lstlisting}[language=C]
typedef struct {
    const char *label;
    const char *appearance;
} alert_button_config;
\end{lstlisting}
\begin{center}
\begin{tabular}{|l|l|>{\raggedright\arraybackslash}p{6cm}|}
\hline
\textbf{Champ} & \textbf{Type} & \textbf{Description} \\
\hline
label & const char* & Texte du bouton (ex: "OK", "Annuler") \\
\hline
appearance & const char* & Style CSS du bouton (ex: "suggested-action") \\
\hline
\end{tabular}
\end{center}
